Paul E. McKenney's 1990 paper [Stochastic Fairness Queuing](https://courses.cs.duke.edu/spring09/cps214/papers/sfq.pdf) contains one of my favorite little distributed algorithms. Stochastic Fairness Queuing is a way to stochastically isolate workloads from different customers in a way that significantly mitigates the effects of noisy neighbors, with O(1) queues and O(1) time.

McKenney starts by describing Fairness Queuing (or *queue per client*):

> This fairness-queuing algorithm operates by maintaining a separate first-come-first-served (FCFS) queue for each conversation. ... Since the queues are serviced in a bit-by-bit round-robin fashion ill-behaved conversations that attempt to use more than their fair share of network resources will face longer delays and larger packet-loss rates than well-behaved conversations that remain within their fair share.

That's a network packet focused view, but the same thing can apply to RPC requests, for example, just by using a different key (e.g. the authorized customer id). The big downside of this in distributed systems is that it requires O(customers) queues, and the related O(customers) work of doing round-robin across those queues.

> Stochastic fairness queuing can be most easily understood by comparing it to strict fairness queuing. The major differences are that the queues are serviced in strict round-robin order and that a simple hash function is used to map from source-destination address pair into a fixed set of queues.

In SFQ, on the other hand, a fixed set of queues is used (so O(1) queues, not O(customers) queues), and customers are assigned to the queues based on a hash. That's great, but still causes the problem of long-term bad luck. If I end up on a queue with a noisy neighbor, I end up there forever.

> If two conversations collide, they will continue to collide, resulting in each conversation of the pair persistently receiving less than its share of bandwidth. This situation is deviated by periodically perturbing the hash function ... so that conversations that collide during one time period are very unlikely to collide during the next.

So noisy neighbors and non-noisy clients move around, preventing somebody from getting bad service for too long.

*SFQ, Shuffle Sharding, and Best-of-2*

We can avoid even those periodic times of bad service by combining three of my favorite small algorithms: SFQ, [shuffle sharding](https://aws.amazon.com/blogs/architecture/shuffle-sharding-massive-and-magical-fault-isolation/), and [best of two](https://brooker.co.za/blog/2012/01/17/two-random.html).

In this variant we:

1. Map each customer to a subset of the queues (that subset could be only two), and
2. For each request put the customer's request in the shortest queue in their subset, and
3. Periodically perturb the subsets.

This approach has absolutely great properties: O(1) queues, O(1) enqueue effort, O(1) dequeue effort, and strong isolation of noisy neighbors from other customers (assuming only a relatively small portion of customers are noisy neighbors).


  <script>
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    
    // Layout constants
    const HASH_BOX = { x: 120, y: 150, w: 100, h: 80 };
    const QUEUE_START_X = 280;
    const QUEUE_WIDTH = 160;
    const QUEUE_HEIGHT = 40;
    const QUEUE_GAP = 52;
    const QUEUE_TOP_Y = 35;
    const NUM_QUEUES = 6;
    const RR_CENTER = { x: 540, y: 190 };
    const RR_RADIUS = 25;
    const PACKET_SIZE = 22;
    const MAX_QUEUE_PACKETS = 6;
    
    // State
    let packets = [];
    let queues = Array(NUM_QUEUES).fill(null).map(() => []);
    let currentRRQueue = 0;
    let paused = false;
    let speed = 1;
    let lastSpawnTime = 0;
    let lastServiceTime = 0;
    let packetIdCounter = 0;
    let burstMode = false;
    let burstCount = 0;
    let timeSinceLastBurst = 0;
    let hashPerturbation = 0; // Changes when perturb button is hit
    
    // Packet colors (by source-dest "conversation")
    const COLORS = ['#e74c3c', '#3498db', '#2ecc71', '#9b59b6', '#f39c12', '#1abc9c', '#e67e22', '#34495e'];
    
    // Precompute queue assignments for each conversation (changes on perturb)
    let conversationToQueue = [];
    function perturbHash() {
      hashPerturbation++;
      // Assign each conversation to a queue based on perturbation
      for (let i = 0; i < COLORS.length; i++) {
        conversationToQueue[i] = (i * 7 + hashPerturbation * 3) % NUM_QUEUES;
      }
    }
    perturbHash(); // Initialize
    
    class Packet {
      constructor() {
        this.id = packetIdCounter++;
        this.conversation = Math.floor(Math.random() * COLORS.length);
        this.color = COLORS[this.conversation];
        this.x = 30;
        this.y = 190;
        this.targetX = this.x;
        this.targetY = this.y;
        this.state = 'entering'; // entering, hashing, toQueue, inQueue, toRR, exiting
        this.queue = -1;
        this.queueSlot = -1;
        this.progress = 0;
      }
      
      hash() {
        // Same conversation (color) always maps to same queue
        return conversationToQueue[this.conversation];
      }
    }
    
    function getQueueY(queueIndex) {
      return QUEUE_TOP_Y + queueIndex * QUEUE_GAP + QUEUE_HEIGHT / 2;
    }
    
    function getQueueSlotX(slot) {
      // Packets line up inside the queue box, from right to left
      const rightEdge = QUEUE_START_X + QUEUE_WIDTH - 8;
      return rightEdge - slot * (PACKET_SIZE + 3) - PACKET_SIZE / 2;
    }
    
    function spawnPacket() {
      const p = new Packet();
      packets.push(p);
    }
    
    function updatePackets(dt) {
      for (let p of packets) {
        const moveSpeed = 250 * speed;
        
        switch (p.state) {
          case 'entering':
            p.targetX = HASH_BOX.x + HASH_BOX.w / 2;
            p.targetY = HASH_BOX.y + HASH_BOX.h / 2;
            break;
            
          case 'hashing':
            p.progress += dt * speed * 3;
            if (p.progress >= 1) {
              p.queue = p.hash();
              p.state = 'toQueue';
              p.progress = 0;
            }
            break;
            
          case 'toQueue':
            // Target the entry point of the queue (left side)
            const qy = getQueueY(p.queue);
            p.targetX = QUEUE_START_X + 20;
            p.targetY = qy;
            break;
            
          case 'inQueue':
            // Packet is in queue, position based on slot
            p.targetX = getQueueSlotX(p.queueSlot);
            p.targetY = getQueueY(p.queue);
            break;
            
          case 'toRR':
            p.targetX = RR_CENTER.x;
            p.targetY = RR_CENTER.y;
            break;
            
          case 'exiting':
            p.targetX = 690;
            p.targetY = RR_CENTER.y;
            break;
        }
        
        // Move toward target
        const dx = p.targetX - p.x;
        const dy = p.targetY - p.y;
        const dist = Math.sqrt(dx * dx + dy * dy);
        
        if (dist > 2) {
          p.x += (dx / dist) * moveSpeed * dt;
          p.y += (dy / dist) * moveSpeed * dt;
        } else {
          p.x = p.targetX;
          p.y = p.targetY;
          
          // State transitions on arrival
          if (p.state === 'entering') {
            p.state = 'hashing';
            p.progress = 0;
          } else if (p.state === 'toQueue') {
            // Add to queue if there's room
            if (queues[p.queue].length < MAX_QUEUE_PACKETS) {
              p.queueSlot = queues[p.queue].length;
              queues[p.queue].push(p);
              p.state = 'inQueue';
            } else {
              // Queue full, drop packet
              p.state = 'dropped';
            }
          } else if (p.state === 'toRR') {
            p.state = 'exiting';
          }
        }
      }
      
      // Remove exited/dropped packets
      packets = packets.filter(p => p.state !== 'dropped' && p.x < 700);
    }
    
    function serviceQueue() {
      // Find next non-empty queue in round-robin
      for (let i = 0; i < NUM_QUEUES; i++) {
        const qIdx = (currentRRQueue + i) % NUM_QUEUES;
        if (queues[qIdx].length > 0) {
          const p = queues[qIdx].shift();
          p.state = 'toRR';
          // Update slots for remaining packets (they shift right)
          queues[qIdx].forEach((pkt, idx) => pkt.queueSlot = idx);
          currentRRQueue = (qIdx + 1) % NUM_QUEUES;
          return;
        }
      }
    }
    
    function drawHashBox() {
      ctx.strokeStyle = 'black';
      ctx.lineWidth = 2;
      ctx.fillStyle = 'white';
      ctx.fillRect(HASH_BOX.x, HASH_BOX.y, HASH_BOX.w, HASH_BOX.h);
      ctx.strokeRect(HASH_BOX.x, HASH_BOX.y, HASH_BOX.w, HASH_BOX.h);
      
      ctx.fillStyle = 'black';
      ctx.font = '12px sans-serif';
      ctx.textAlign = 'center';
      ctx.fillText('Distribute', HASH_BOX.x + HASH_BOX.w/2, HASH_BOX.y + 25);
      ctx.fillText('Packets By', HASH_BOX.x + HASH_BOX.w/2, HASH_BOX.y + 40);
      ctx.fillText('Hash on', HASH_BOX.x + HASH_BOX.w/2, HASH_BOX.y + 55);
      ctx.fillText('Address', HASH_BOX.x + HASH_BOX.w/2, HASH_BOX.y + 70);
    }
    
    function drawQueues() {
      ctx.strokeStyle = 'black';
      ctx.lineWidth = 2;
      ctx.font = '11px sans-serif';
      ctx.textAlign = 'left';
      
      for (let i = 0; i < NUM_QUEUES; i++) {
        const y = QUEUE_TOP_Y + i * QUEUE_GAP;
        ctx.fillStyle = '#fafafa';
        ctx.fillRect(QUEUE_START_X, y, QUEUE_WIDTH, QUEUE_HEIGHT);
        ctx.strokeRect(QUEUE_START_X, y, QUEUE_WIDTH, QUEUE_HEIGHT);
        
        // Label on left side
        ctx.fillStyle = '#666';
        ctx.fillText('FCFS Queue ' + i, QUEUE_START_X + 5, y + QUEUE_HEIGHT/2 + 4);
      }
    }
    
    function drawRoundRobin() {
      ctx.strokeStyle = 'black';
      ctx.lineWidth = 2;
      ctx.fillStyle = 'white';
      ctx.beginPath();
      ctx.arc(RR_CENTER.x, RR_CENTER.y, RR_RADIUS, 0, Math.PI * 2);
      ctx.fill();
      ctx.stroke();
      
      ctx.fillStyle = 'black';
      ctx.font = '10px sans-serif';
      ctx.textAlign = 'center';
      ctx.fillText('Round', RR_CENTER.x, RR_CENTER.y - 3);
      ctx.fillText('Robin', RR_CENTER.x, RR_CENTER.y + 9);
      
      // Draw pointer indicating current queue
      const angle = -Math.PI/2 + (currentRRQueue / NUM_QUEUES) * Math.PI * 2;
      ctx.strokeStyle = '#e74c3c';
      ctx.lineWidth = 3;
      ctx.beginPath();
      ctx.moveTo(RR_CENTER.x, RR_CENTER.y);
      ctx.lineTo(
        RR_CENTER.x + Math.cos(angle) * (RR_RADIUS - 5),
        RR_CENTER.y + Math.sin(angle) * (RR_RADIUS - 5)
      );
      ctx.stroke();
      ctx.lineWidth = 2;
      ctx.strokeStyle = 'black';
    }
    
    function drawConnections() {
      ctx.strokeStyle = '#999';
      ctx.lineWidth = 1.5;
      
      // Input arrow
      ctx.beginPath();
      ctx.moveTo(20, 190);
      ctx.lineTo(HASH_BOX.x - 5, 190);
      ctx.stroke();
      drawArrowHead(HASH_BOX.x - 5, 190, 0);
      
      ctx.font = '10px sans-serif';
      ctx.textAlign = 'center';
      ctx.fillStyle = 'black';
      ctx.fillText('Input', 55, 175);
      ctx.fillText('Packets', 55, 187);
      
      // Hash to queues
      for (let i = 0; i < NUM_QUEUES; i++) {
        const qy = getQueueY(i);
        ctx.beginPath();
        ctx.moveTo(HASH_BOX.x + HASH_BOX.w, HASH_BOX.y + HASH_BOX.h/2);
        ctx.lineTo(QUEUE_START_X - 5, qy);
        ctx.stroke();
      }
      
      // Queues to RR
      for (let i = 0; i < NUM_QUEUES; i++) {
        const qy = getQueueY(i);
        ctx.beginPath();
        ctx.moveTo(QUEUE_START_X + QUEUE_WIDTH, qy);
        ctx.lineTo(RR_CENTER.x - RR_RADIUS - 5, RR_CENTER.y);
        ctx.stroke();
      }
      
      // Output arrow
      ctx.beginPath();
      ctx.moveTo(RR_CENTER.x + RR_RADIUS, RR_CENTER.y);
      ctx.lineTo(680, RR_CENTER.y);
      ctx.stroke();
      drawArrowHead(680, RR_CENTER.y, 0);
      
      ctx.fillText('Output', 640, 175);
      ctx.fillText('Packets', 640, 187);
    }
    
    function drawArrowHead(x, y, angle) {
      ctx.fillStyle = '#999';
      ctx.beginPath();
      ctx.moveTo(x, y);
      ctx.lineTo(x - 10 * Math.cos(angle - 0.4), y - 10 * Math.sin(angle - 0.4));
      ctx.lineTo(x - 10 * Math.cos(angle + 0.4), y - 10 * Math.sin(angle + 0.4));
      ctx.closePath();
      ctx.fill();
    }
    
    function drawPackets() {
      for (let p of packets) {
        ctx.fillStyle = p.color;
        ctx.strokeStyle = '#333';
        ctx.lineWidth = 1.5;
        
        // Draw packet as small square
        const halfSize = PACKET_SIZE / 2;
        ctx.fillRect(p.x - halfSize, p.y - halfSize, PACKET_SIZE, PACKET_SIZE);
        ctx.strokeRect(p.x - halfSize, p.y - halfSize, PACKET_SIZE, PACKET_SIZE);
        
        // Show hash animation
        if (p.state === 'hashing') {
          ctx.fillStyle = 'rgba(0,0,0,0.8)';
          ctx.font = 'bold 14px monospace';
          ctx.textAlign = 'center';
          ctx.fillText('→ Q' + p.hash(), p.x, p.y - 18);
        }
      }
    }
    
    function drawStats() {
      ctx.fillStyle = '#333';
      ctx.font = '11px sans-serif';
      ctx.textAlign = 'left';
      
      let totalInQueues = queues.reduce((sum, q) => sum + q.length, 0);
      ctx.fillText('Packets in queues: ' + totalInQueues, 10, 390);
      ctx.fillText('Current RR queue: ' + currentRRQueue, 150, 390);
    }
    
    function draw() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      
      drawConnections();
      drawHashBox();
      drawQueues();
      drawRoundRobin();
      drawPackets();
      drawStats();
    }
    
    let lastTime = 0;
    function gameLoop(timestamp) {
      const dt = Math.min((timestamp - lastTime) / 1000, 0.1);
      lastTime = timestamp;
      
      if (!paused) {
        // Bursty arrival pattern
        timeSinceLastBurst += dt * speed;
        
        if (burstMode) {
          // In burst: spawn packets rapidly
          lastSpawnTime += dt * speed;
          if (lastSpawnTime > 0.12) {
            spawnPacket();
            lastSpawnTime = 0;
            burstCount++;
            if (burstCount >= 8 + Math.floor(Math.random() * 8)) {
              // End burst
              burstMode = false;
              burstCount = 0;
              timeSinceLastBurst = 0;
            }
          }
        } else {
          // Between bursts: occasional single packets, then start new burst
          if (timeSinceLastBurst > 1.5 + Math.random() * 1.5) {
            burstMode = true;
            burstCount = 0;
          } else if (Math.random() < 0.02 * speed) {
            // Occasional packet between bursts
            spawnPacket();
          }
        }
        
        // Service queues (slower than arrival during bursts to build up queues)
        lastServiceTime += dt * speed;
        if (lastServiceTime > 0.35) {
          serviceQueue();
          lastServiceTime = 0;
        }
        
        updatePackets(dt);
      }
      
      draw();
      requestAnimationFrame(gameLoop);
    }
    
    // Controls
    document.getElementById('pauseBtn').addEventListener('click', () => {
      paused = !paused;
      document.getElementById('pauseBtn').textContent = paused ? 'Resume' : 'Pause';
    });
    
    document.getElementById('speedBtn').addEventListener('click', () => {
      if (speed === 1) {
        speed = 2;
        document.getElementById('speedBtn').textContent = 'Speed: Fast';
      } else if (speed === 2) {
        speed = 0.5;
        document.getElementById('speedBtn').textContent = 'Speed: Slow';
      } else {
        speed = 1;
        document.getElementById('speedBtn').textContent = 'Speed: Normal';
      }
    });
    
    document.getElementById('resetBtn').addEventListener('click', () => {
      packets = [];
      queues = Array(NUM_QUEUES).fill(null).map(() => []);
      currentRRQueue = 0;
      lastSpawnTime = 0;
      lastServiceTime = 0;
      burstMode = false;
      burstCount = 0;
      timeSinceLastBurst = 0;
    });
    
    document.getElementById('perturbBtn').addEventListener('click', () => {
      perturbHash();
    });
    
    // Start
    requestAnimationFrame(gameLoop);
  </script>
