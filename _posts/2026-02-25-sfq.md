Paul E. McKenney's 1990 paper [Stochastic Fairness Queuing](https://courses.cs.duke.edu/spring09/cps214/papers/sfq.pdf) contains one of my favorite little distributed algorithms. Stochastic Fairness Queuing is a way to stochastically isolate workloads from different customers in a way that significantly mitigates the effects of noisy neighbors, with O(1) queues and O(1) time.

McKenney starts by describing Fairness Queuing (or *queue per client*):

> This fairness-queuing algorithm operates by maintaining a separate first-come-first-served (FCFS) queue for each conversation. ... Since the queues are serviced in a bit-by-bit round-robin fashion ill-behaved conversations that attempt to use more than their fair share of network resources will face longer delays and larger packet-loss rates than well-behaved conversations that remain within their fair share.

That's a network packet focused view, but the same thing can apply to RPC requests, for example, just by using a different key (e.g. the authorized customer id). The big downside of this in distributed systems is that it requires O(customers) queues, and the related O(customers) work of doing round-robin across those queues.

> Stochastic fairness queuing can be most easily understood by comparing it to strict fairness queuing. The major differences are that the queues are serviced in strict round-robin order and that a simple hash function is used to map from source-destination address pair into a fixed set of queues.

In SFQ, on the other hand, a fixed set of queues is used (so O(1) queues, not O(customers) queues), and customers are assigned to the queues based on a hash. That's great, but still causes the problem of long-term bad luck. If I end up on a queue with a noisy neighbor, I end up there forever.

> If two conversations collide, they will continue to collide, resulting in each conversation of the pair persistently receiving less than its share of bandwidth. This situation is deviated by periodically perturbing the hash function ... so that conversations that collide during one time period are very unlikely to collide during the next.

So noisy neighbors and non-noisy clients move around, preventing somebody from getting bad service for too long.

*SFQ, Shuffle Sharding, and Best-of-2*

We can avoid even those periodic times of bad service by combining three of my favorite small algorithms: SFQ, [shuffle sharding](https://aws.amazon.com/blogs/architecture/shuffle-sharding-massive-and-magical-fault-isolation/), and [best of two](https://brooker.co.za/blog/2012/01/17/two-random.html).

In this variant we:

1. Map each customer to a subset of the queues (that subset could be only two), and
2. For each request put the customer's request in the shortest queue in their subset, and
3. Periodically perturb the subsets.

This approach has absolutely great properties: O(1) queues, O(1) enqueue effort, O(1) dequeue effort, and strong isolation of noisy neighbors from other customers (assuming only a relatively small portion of customers are noisy neighbors).

  <script>
    // Shared constants
    const COLORS = ['#e74c3c', '#3498db', '#2ecc71', '#9b59b6', '#f39c12', '#1abc9c', '#e67e22', '#34495e'];
    const NUM_QUEUES = 6;
    const MAX_QUEUE_PACKETS = 6;
    const PACKET_SIZE = 20;
    
    // Layout constants
    const HASH_BOX = { x: 100, y: 140, w: 90, h: 75 };
    const QUEUE_START_X = 240;
    const QUEUE_WIDTH = 140;
    const QUEUE_HEIGHT = 38;
    const QUEUE_GAP = 50;
    const QUEUE_TOP_Y = 30;
    const RR_CENTER = { x: 470, y: 180 };
    const RR_RADIUS = 22;
    
    // Shared state
    let paused = false;
    let speed = 1;
    let hashPerturbation = 0;
    
    // Shared timing for synchronized packet generation
    let lastSpawnTime = 0;
    let burstMode = false;
    let burstCount = 0;
    let timeSinceLastBurst = 0;
    let pendingPackets = []; // Queue of conversation IDs to spawn
    
    // Hash mappings (shared between both simulations)
    let conversationToQueue1 = [];
    let conversationToQueue2 = [];
    
    function perturbHash() {
      hashPerturbation++;
      for (let i = 0; i < COLORS.length; i++) {
        conversationToQueue1[i] = (i * 7 + hashPerturbation * 3) % NUM_QUEUES;
        let q2 = (i * 11 + hashPerturbation * 5 + 3) % NUM_QUEUES;
        if (q2 === conversationToQueue1[i]) {
          q2 = (q2 + 1) % NUM_QUEUES;
        }
        conversationToQueue2[i] = q2;
      }
    }
    perturbHash();
    
    function getQueueY(queueIndex) {
      return QUEUE_TOP_Y + queueIndex * QUEUE_GAP + QUEUE_HEIGHT / 2;
    }
    
    function getQueueSlotX(slot) {
      const rightEdge = QUEUE_START_X + QUEUE_WIDTH - 6;
      return rightEdge - slot * (PACKET_SIZE + 2) - PACKET_SIZE / 2;
    }
    
    function generateConversation() {
      const rand = Math.random();
      if (rand < 0.4) {
        return 0; // Red - noisy neighbor
      } else {
        return 1 + Math.floor(Math.random() * (COLORS.length - 1));
      }
    }
    
    // Simulation class - handles one instance of the visualization
    class Simulation {
      constructor(canvasId, twoChoice) {
        this.canvas = document.getElementById(canvasId);
        this.ctx = this.canvas.getContext('2d');
        this.twoChoice = twoChoice;
        this.packets = [];
        this.queues = Array(NUM_QUEUES).fill(null).map(() => []);
        this.currentRRQueue = 0;
        this.lastServiceTime = 0;
        this.packetIdCounter = 0;
      }
      
      createPacket(conversation) {
        const p = {
          id: this.packetIdCounter++,
          conversation: conversation,
          color: COLORS[conversation],
          x: 25,
          y: 175,
          targetX: 25,
          targetY: 175,
          state: 'entering',
          queue: -1,
          queueSlot: -1,
          progress: 0,
          hashChoice1: -1,
          hashChoice2: -1,
          chosenQueue: -1
        };
        this.packets.push(p);
      }
      
      computeHash(p) {
        p.hashChoice1 = conversationToQueue1[p.conversation];
        if (this.twoChoice) {
          p.hashChoice2 = conversationToQueue2[p.conversation];
          const len1 = this.queues[p.hashChoice1].length;
          const len2 = this.queues[p.hashChoice2].length;
          p.chosenQueue = len1 <= len2 ? p.hashChoice1 : p.hashChoice2;
        } else {
          p.chosenQueue = p.hashChoice1;
        }
        return p.chosenQueue;
      }
      
      update(dt) {
        const moveSpeed = 250 * speed;
        
        for (let p of this.packets) {
          switch (p.state) {
            case 'entering':
              p.targetX = HASH_BOX.x + HASH_BOX.w / 2;
              p.targetY = HASH_BOX.y + HASH_BOX.h / 2;
              break;
            case 'hashing':
              p.progress += dt * speed * 3;
              if (p.progress >= 1) {
                p.queue = p.chosenQueue;
                p.state = 'toQueue';
                p.progress = 0;
              }
              break;
            case 'toQueue':
              p.targetX = QUEUE_START_X + 15;
              p.targetY = getQueueY(p.queue);
              break;
            case 'inQueue':
              p.targetX = getQueueSlotX(p.queueSlot);
              p.targetY = getQueueY(p.queue);
              break;
            case 'toRR':
              p.targetX = RR_CENTER.x;
              p.targetY = RR_CENTER.y;
              break;
            case 'exiting':
              p.targetX = 570;
              p.targetY = RR_CENTER.y;
              break;
          }
          
          const dx = p.targetX - p.x;
          const dy = p.targetY - p.y;
          const dist = Math.sqrt(dx * dx + dy * dy);
          
          if (dist > 2) {
            p.x += (dx / dist) * moveSpeed * dt;
            p.y += (dy / dist) * moveSpeed * dt;
          } else {
            p.x = p.targetX;
            p.y = p.targetY;
            
            if (p.state === 'entering') {
              p.state = 'hashing';
              p.progress = 0;
              this.computeHash(p);
            } else if (p.state === 'toQueue') {
              if (this.queues[p.queue].length < MAX_QUEUE_PACKETS) {
                p.queueSlot = this.queues[p.queue].length;
                this.queues[p.queue].push(p);
                p.state = 'inQueue';
              } else {
                p.state = 'dropped';
              }
            } else if (p.state === 'toRR') {
              p.state = 'exiting';
            }
          }
        }
        
        this.packets = this.packets.filter(p => p.state !== 'dropped' && p.x < 580);
        
        // Service queues
        this.lastServiceTime += dt * speed;
        if (this.lastServiceTime > 0.35) {
          this.serviceQueue();
          this.lastServiceTime = 0;
        }
      }
      
      serviceQueue() {
        for (let i = 0; i < NUM_QUEUES; i++) {
          const qIdx = (this.currentRRQueue + i) % NUM_QUEUES;
          if (this.queues[qIdx].length > 0) {
            const p = this.queues[qIdx].shift();
            p.state = 'toRR';
            this.queues[qIdx].forEach((pkt, idx) => pkt.queueSlot = idx);
            this.currentRRQueue = (qIdx + 1) % NUM_QUEUES;
            return;
          }
        }
      }
      
      draw() {
        const ctx = this.ctx;
        ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
        
        // Draw connections
        ctx.strokeStyle = '#999';
        ctx.lineWidth = 1.5;
        
        // Input arrow
        ctx.beginPath();
        ctx.moveTo(15, 175);
        ctx.lineTo(HASH_BOX.x - 5, 175);
        ctx.stroke();
        this.drawArrowHead(HASH_BOX.x - 5, 175, 0);
        
        ctx.font = '9px sans-serif';
        ctx.textAlign = 'center';
        ctx.fillStyle = 'black';
        ctx.fillText('Input', 45, 162);
        ctx.fillText('Packets', 45, 173);
        
        // Hash to queues
        for (let i = 0; i < NUM_QUEUES; i++) {
          const qy = getQueueY(i);
          ctx.beginPath();
          ctx.moveTo(HASH_BOX.x + HASH_BOX.w, HASH_BOX.y + HASH_BOX.h/2);
          ctx.lineTo(QUEUE_START_X - 5, qy);
          ctx.stroke();
        }
        
        // Queues to RR
        for (let i = 0; i < NUM_QUEUES; i++) {
          const qy = getQueueY(i);
          ctx.beginPath();
          ctx.moveTo(QUEUE_START_X + QUEUE_WIDTH, qy);
          ctx.lineTo(RR_CENTER.x - RR_RADIUS - 5, RR_CENTER.y);
          ctx.stroke();
        }
        
        // Output arrow
        ctx.beginPath();
        ctx.moveTo(RR_CENTER.x + RR_RADIUS, RR_CENTER.y);
        ctx.lineTo(560, RR_CENTER.y);
        ctx.stroke();
        this.drawArrowHead(560, RR_CENTER.y, 0);
        
        ctx.fillText('Output', 530, 167);
        ctx.fillText('Packets', 530, 178);
        
        // Draw hash box
        ctx.strokeStyle = 'black';
        ctx.lineWidth = 2;
        ctx.fillStyle = 'white';
        ctx.fillRect(HASH_BOX.x, HASH_BOX.y, HASH_BOX.w, HASH_BOX.h);
        ctx.strokeRect(HASH_BOX.x, HASH_BOX.y, HASH_BOX.w, HASH_BOX.h);
        
        ctx.fillStyle = 'black';
        ctx.font = '11px sans-serif';
        ctx.textAlign = 'center';
        ctx.fillText('Distribute', HASH_BOX.x + HASH_BOX.w/2, HASH_BOX.y + 20);
        ctx.fillText('Packets By', HASH_BOX.x + HASH_BOX.w/2, HASH_BOX.y + 33);
        ctx.fillText('Hash on', HASH_BOX.x + HASH_BOX.w/2, HASH_BOX.y + 46);
        ctx.fillText('Address', HASH_BOX.x + HASH_BOX.w/2, HASH_BOX.y + 59);
        
        // Draw queues
        ctx.strokeStyle = 'black';
        ctx.lineWidth = 2;
        ctx.font = '10px sans-serif';
        ctx.textAlign = 'left';
        
        for (let i = 0; i < NUM_QUEUES; i++) {
          const y = QUEUE_TOP_Y + i * QUEUE_GAP;
          ctx.fillStyle = '#fafafa';
          ctx.fillRect(QUEUE_START_X, y, QUEUE_WIDTH, QUEUE_HEIGHT);
          ctx.strokeRect(QUEUE_START_X, y, QUEUE_WIDTH, QUEUE_HEIGHT);
          ctx.fillStyle = '#666';
          ctx.fillText('FCFS Q' + i, QUEUE_START_X + 4, y + QUEUE_HEIGHT/2 + 3);
        }
        
        // Draw round robin
        ctx.strokeStyle = 'black';
        ctx.lineWidth = 2;
        ctx.fillStyle = 'white';
        ctx.beginPath();
        ctx.arc(RR_CENTER.x, RR_CENTER.y, RR_RADIUS, 0, Math.PI * 2);
        ctx.fill();
        ctx.stroke();
        
        ctx.fillStyle = 'black';
        ctx.font = '9px sans-serif';
        ctx.textAlign = 'center';
        ctx.fillText('Round', RR_CENTER.x, RR_CENTER.y - 2);
        ctx.fillText('Robin', RR_CENTER.x, RR_CENTER.y + 8);
        
        const angle = -Math.PI/2 + (this.currentRRQueue / NUM_QUEUES) * Math.PI * 2;
        ctx.strokeStyle = '#e74c3c';
        ctx.lineWidth = 3;
        ctx.beginPath();
        ctx.moveTo(RR_CENTER.x, RR_CENTER.y);
        ctx.lineTo(
          RR_CENTER.x + Math.cos(angle) * (RR_RADIUS - 5),
          RR_CENTER.y + Math.sin(angle) * (RR_RADIUS - 5)
        );
        ctx.stroke();
        
        // Draw packets
        for (let p of this.packets) {
          ctx.fillStyle = p.color;
          ctx.strokeStyle = '#333';
          ctx.lineWidth = 1.5;
          
          const halfSize = PACKET_SIZE / 2;
          ctx.fillRect(p.x - halfSize, p.y - halfSize, PACKET_SIZE, PACKET_SIZE);
          ctx.strokeRect(p.x - halfSize, p.y - halfSize, PACKET_SIZE, PACKET_SIZE);
          
          if (p.state === 'hashing') {
            ctx.fillStyle = 'rgba(0,0,0,0.8)';
            ctx.font = 'bold 10px monospace';
            ctx.textAlign = 'center';
            if (this.twoChoice) {
              const q1Len = this.queues[p.hashChoice1].length;
              const q2Len = this.queues[p.hashChoice2].length;
              ctx.fillText('Q' + p.hashChoice1 + '(' + q1Len + ') vs Q' + p.hashChoice2 + '(' + q2Len + ')', p.x, p.y - 22);
            }
            ctx.fillText('→ Q' + p.chosenQueue, p.x, p.y - 10);
          }
        }
        
        // Draw stats
        ctx.fillStyle = '#333';
        ctx.font = '10px sans-serif';
        ctx.textAlign = 'left';
        let totalInQueues = this.queues.reduce((sum, q) => sum + q.length, 0);
        ctx.fillText('In queues: ' + totalInQueues, 10, 370);
      }
      
      drawArrowHead(x, y, angle) {
        this.ctx.fillStyle = '#999';
        this.ctx.beginPath();
        this.ctx.moveTo(x, y);
        this.ctx.lineTo(x - 8 * Math.cos(angle - 0.4), y - 8 * Math.sin(angle - 0.4));
        this.ctx.lineTo(x - 8 * Math.cos(angle + 0.4), y - 8 * Math.sin(angle + 0.4));
        this.ctx.closePath();
        this.ctx.fill();
      }
      
      reset() {
        this.packets = [];
        this.queues = Array(NUM_QUEUES).fill(null).map(() => []);
        this.currentRRQueue = 0;
        this.lastServiceTime = 0;
      }
    }
    
    // Create both simulations
    const sim1 = new Simulation('canvas1', false); // Single hash
    const sim2 = new Simulation('canvas2', true);  // Two-choice
    
    let lastTime = 0;
    function gameLoop(timestamp) {
      const dt = Math.min((timestamp - lastTime) / 1000, 0.1);
      lastTime = timestamp;
      
      if (!paused) {
        // Shared bursty arrival pattern
        timeSinceLastBurst += dt * speed;
        
        if (burstMode) {
          lastSpawnTime += dt * speed;
          if (lastSpawnTime > 0.12) {
            const conv = generateConversation();
            sim1.createPacket(conv);
            sim2.createPacket(conv);
            lastSpawnTime = 0;
            burstCount++;
            if (burstCount >= 8 + Math.floor(Math.random() * 8)) {
              burstMode = false;
              burstCount = 0;
              timeSinceLastBurst = 0;
            }
          }
        } else {
          if (timeSinceLastBurst > 1.5 + Math.random() * 1.5) {
            burstMode = true;
            burstCount = 0;
          } else if (Math.random() < 0.02 * speed) {
            const conv = generateConversation();
            sim1.createPacket(conv);
            sim2.createPacket(conv);
          }
        }
        
        sim1.update(dt);
        sim2.update(dt);
      }
      
      sim1.draw();
      sim2.draw();
      
      requestAnimationFrame(gameLoop);
    }
    
    // Controls
    document.getElementById('pauseBtn').addEventListener('click', () => {
      paused = !paused;
      document.getElementById('pauseBtn').textContent = paused ? 'Resume' : 'Pause';
    });
    
    document.getElementById('speedBtn').addEventListener('click', () => {
      if (speed === 1) {
        speed = 2;
        document.getElementById('speedBtn').textContent = 'Speed: Fast';
      } else if (speed === 2) {
        speed = 0.5;
        document.getElementById('speedBtn').textContent = 'Speed: Slow';
      } else {
        speed = 1;
        document.getElementById('speedBtn').textContent = 'Speed: Normal';
      }
    });
    
    document.getElementById('perturbBtn').addEventListener('click', () => {
      perturbHash();
    });
    
    document.getElementById('resetBtn').addEventListener('click', () => {
      sim1.reset();
      sim2.reset();
      lastSpawnTime = 0;
      burstMode = false;
      burstCount = 0;
      timeSinceLastBurst = 0;
    });
    
    requestAnimationFrame(gameLoop);
  </script>
